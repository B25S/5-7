导航栏目

暂时使用的是链接标签，应该还有更好的方式，更符合html5的方式。

从后端拿json数据来动态的加载目录。

	json数据(Navigator.txt文件存储):
		
			[
				{1:"0x0001"},
				{2:"0x0002"},
				{3:"0x0003"},
				{4:"0x0004"},
				{5:"0x0005"},
				{6:"0x0006"}
			]
		
	JS使用ajax调用代码(导航栏id为Navigator):
		
                var xmlhttp = new XMLHttpRequest();
                //onreadystatechange为一个未设置的回调函数，在XMLHttpRequest.open()请求完毕时候调用
                xmlhttp.onreadystatechange=function(){
                    if(xmlhttp.readyState==4 && xmlhttp.status==200){
                        //设置目录布局
                        //这里可以尝试使用json来格式化数据,XMLHttpRequest.response是一个object对象
                        var nav_con = eval(xmlhttp.response);
                        var navigator = document.getElementById("Navigator");
                        for (var i = 0; i < nav_con.length; i++){
                            //创建tag类型的DOM元素
                            var link = document.createElement("a");
                            //创建文本类型DOM元素
                            var text = document.createTextNode(" | ");
                            if(i != 0) {
                                navigator.appendChild(text);
                            }
                            link.text = nav_con[i][eval(i+1)];
                            link.setAttribute("href","#");
                            navigator.appendChild(link);
                        }
                        i = 0;
                    }
                }
                xmlhttp.open("get","Navigator.txt",true);
                xmlhttp.send();
